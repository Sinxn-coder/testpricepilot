async function getOptimizedPrice(base, country) {
  const res = await fetch('/calculate-price', {
    method: 'POST',
    headers: { 
      'Authorization': 'Bearer YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      base_price: base,
      country: country,
      currency: 'USD',
      min_margin: 0.2
    })
  });

  return await res.json();
}
